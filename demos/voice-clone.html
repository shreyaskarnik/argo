Step 3

One config change.

Point your Argo config at the mlx-audio engine with your reference clip. Everything else stays the same.

argo.config.mjs 
import { defineConfig, engines } from '@argo-video/cli';

export default defineConfig({
  tts: {
    engine: engines.mlxAudio({
      model: 'mlx-community/Qwen3-TTS-12Hz-0.6B-Base-bf16',
      refAudio: './assets/ref-voice.wav',
      refText: 'Hi, my name is Shreyas. I build developer tools.',
    }),
  },
  video: { browser: 'webkit' },
});
